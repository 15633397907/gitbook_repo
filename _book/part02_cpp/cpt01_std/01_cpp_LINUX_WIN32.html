Section1.1 WIN32 -> LINUX

LINUX & WIN32

头文件替换

library windows linux

dirent #include #include

sprintf_s -> snprintf

char tag[256];
char* str;
#ifdef windows
        sprintf_s(tag,"<%s>",str);
#elif defined __linux__
        snprintf(tag,256,"<%s>",str);
#endif

fopen_s -> fopen

const char* charPath
FILE *pFile
#ifdef windows
fopen_s(&pFile,charPath,"rb");
#elif defined __linux__
pFile = fopen(charPath,"rb");
#endif

_strlwe(str);

*ptr = tolower(*ptr);
}
#endif
