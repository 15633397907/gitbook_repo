## sprintf_s -> snprintf

```c++
char tag[256];
char* str;
#ifdef windows
        sprintf_s(tag,"<%s>",str);
#elif defined __linux__
        snprintf(tag,256,"<%s>",str);
#endif
```

## fopen_s -> fopen

```C++
const char* charPath
FILE *pFile
#ifdef windows
fopen_s(&pFile,charPath,"rb");
#elif defined __linux__
pFile = fopen(charPath,"rb");
#endif
```

## _strlwr 平替（无函数）

```C++
char str[256];
 #ifdef windows
_strlwe(str);
#elif defined __linux__
for(char* ptr = str; *ptr; ptr++){
*ptr = tolower(*ptr);
}
#endif
```

## CPU数量

openmp需要使用

```C++
int intnCPUs;
#ifdef windows
#include<windows.h>
SYSTEM_INFO SystemInfo;
GetSystemInfo(&SystemInfo);
intnCPUs = SystemInfo.dwNumberOfProcessors;
#elif defined __linux__
#include <unistd.h>
#include <iostream>
intnCPUs = sysconf(_SC_NPROCESSORS_CONF);
#endif
intnCPUs *= 0.9;
```
